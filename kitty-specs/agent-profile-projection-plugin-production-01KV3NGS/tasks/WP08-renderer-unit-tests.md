```python
# tests/specify_cli/tool_surface/test_profile_renderers.py
import pytest
from pathlib import Path
from specify_cli.tool_surface.profiles.renderers import ClaudeCodeProfileRenderer

def _make_profile(profile_id="analyst-core", **kwargs):
    from specify_cli.tool_surface.profiles.models import AgentProfile
    return AgentProfile(
        profile_id=profile_id,
        friendly_name=kwargs.get("friendly_name", "Analyst"),
        description=kwargs.get("description", "A core analyst profile."),
        system_prompt=kwargs.get("system_prompt", "You are a careful analyst."),
    )

class TestClaudeCodeProfileRenderer:
    def setup_method(self):
        self.renderer = ClaudeCodeProfileRenderer()
        self.profile = _make_profile()

    def test_output_path(self, tmp_path):
        path = self.renderer.output_path(tmp_path, self.profile)
        assert str(path).endswith(".claude/agents/analyst-core.md")
        assert path.parent == tmp_path / ".claude" / "agents"

    def test_yaml_frontmatter_has_name_and_description(self):
        rendered = self.renderer.render(self.profile)
        text = rendered if isinstance(rendered, str) else rendered.decode()
        assert "name:" in text
        assert "description:" in text

    def test_provenance_footer(self):
        rendered = self.renderer.render(self.profile)
        text = rendered if isinstance(rendered, str) else rendered.decode()
        assert "generated by spec-kitty" in text

    def test_idempotent_re_render(self):
        first = self.renderer.render(self.profile)
        second = self.renderer.render(self.profile)
        assert first == second

    def test_can_render_own_format_key(self):
        assert self.renderer.can_render("claude")
        assert self.renderer.can_render(self.renderer.format_key)
        assert not self.renderer.can_render("codex")
```

### T036 — Unit tests for `CodexProfileRenderer`

```python
# tests/specify_cli/tool_surface/test_codex_renderer.py
import tomllib
import pytest
from pathlib import Path
from specify_cli.tool_surface.profiles.codex_renderer import CodexProfileRenderer

class TestCodexProfileRenderer:
    def setup_method(self):
        self.renderer = CodexProfileRenderer()

    def test_output_path(self, tmp_path):
        profile = _make_profile()
        path = self.renderer.output_path(tmp_path, profile)
        assert str(path).endswith(".codex/agents/analyst-core.toml")

    def test_renders_valid_toml(self):
        profile = _make_profile()
        rendered = self.renderer.render(profile)
        text = rendered.decode() if isinstance(rendered, bytes) else rendered
        doc = tomllib.loads(text)
        assert isinstance(doc, dict)

    def test_required_fields_present(self):
        profile = _make_profile()
        rendered = self.renderer.render(profile)
        text = rendered.decode() if isinstance(rendered, bytes) else rendered
        doc = tomllib.loads(text)
        assert "name" in doc
        assert "description" in doc
        assert "developer_instructions" in doc

    def test_optional_field_passthrough_when_present(self):
        profile = _make_profile()
        profile.model = "claude-sonnet-4-6"
        rendered = self.renderer.render(profile)
        text = rendered.decode() if isinstance(rendered, bytes) else rendered
        doc = tomllib.loads(text)
        assert "model" in doc
        assert doc["model"] == "claude-sonnet-4-6"

    def test_optional_field_absent_when_not_set(self):
        profile = _make_profile()
        rendered = self.renderer.render(profile)
        text = rendered.decode() if isinstance(rendered, bytes) else rendered
        doc = tomllib.loads(text)
        assert "model" not in doc
        assert "sandbox_mode" not in doc

    def test_can_render_codex_tool_keys(self):
        assert self.renderer.can_render("codex")
        assert self.renderer.can_render("codex-cli")
        assert self.renderer.can_render(self.renderer.format_key)
        assert not self.renderer.can_render("claude")
```

### T038 — Parametric test covering all three renderers

```python
@pytest.mark.parametrize("renderer_cls,tool_key", [
    (ClaudeCodeProfileRenderer, "claude"),
    (CodexProfileRenderer, "codex"),
    (CopilotProfileRenderer, "copilot"),
])
def test_renderer_parametric(renderer_cls, tool_key, tmp_path):
    renderer = renderer_cls()
    profile = _make_profile()

    # can_render() must accept own tool_key
    assert renderer.can_render(tool_key), f"{renderer_cls.__name__}.can_render({tool_key!r}) returned False"

    # output_path() must return a non-None Path under tmp_path or home
    path = renderer.output_path(tmp_path, profile)
    assert path is not None
    assert isinstance(path, Path)

    # render() must return non-empty output
    rendered = renderer.render(profile)
    assert rendered, f"{renderer_cls.__name__}.render() returned empty output"
    text = rendered.decode() if isinstance(rendered, bytes) else rendered
    assert len(text) > 50, f"Rendered output suspiciously short: {len(text)} chars"
```
