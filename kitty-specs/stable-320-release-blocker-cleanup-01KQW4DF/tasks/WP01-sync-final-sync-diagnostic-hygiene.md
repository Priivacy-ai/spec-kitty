## Objective

Fix GitHub issue **#952**: successful local lifecycle commands leak SaaS final-sync error messages into their output (including stdout in `--json` mode), displaying red failure text that contradicts a successful exit code.

**What you will deliver**:
1. A new module `src/specify_cli/sync/diagnostics.py` that owns all final-sync stderr diagnostic emission — deduplicated, categorized, non-fatal.
2. Refactored `sync/daemon.py` and `sync/batch.py` that route final-sync errors through the new module instead of writing directly to the console.
3. Six regression tests in `tests/sync/test_final_sync_diagnostics.py`.
4. Extended assertions in `tests/e2e/test_mission_create_clean_output.py`.

---

## Context

During the 3.2.0 P1 Release Confidence smoke run, lifecycle commands that completed successfully (exit 0, local state mutated correctly) simultaneously emitted:
- Red `Connection failed` text on output
- `sync.final_sync_lock_unavailable` messages
- `Another spec-kitty process is refreshing the auth session; retry in a moment`

These messages come from `sync/daemon.py` (fcntl lock acquisition timeout) and `sync/batch.py` (error categorization + formatting). They are not fatal — the local mutation succeeded — but they make the command appear to have failed.

The fix is behavioral: route all non-fatal final-sync error output to stderr only, using a single, deduplicating diagnostic function. No SaaS-side changes are needed.

**Key files to understand before editing**:
- `src/specify_cli/sync/daemon.py`: look for `DaemonStartOutcome`, `skipped_reason`, and any `console.print(...)` calls related to sync lock failure.
- `src/specify_cli/sync/batch.py`: look for `categorize_error()`, `format_sync_summary()`, `generate_failure_report()`, and any direct console/print calls for final-sync errors.

## Subtask T001 — Create `src/specify_cli/sync/diagnostics.py`

**Purpose**: Single authority for all final-sync diagnostic emission to stderr. This module is the entire fix for the "where do errors go" problem.

**Create** `src/specify_cli/sync/diagnostics.py`:

```python
"""
Final-sync diagnostic emission and classification.

This module is the sole write path for non-fatal final-sync failure diagnostics.
All error signals from daemon.py and batch.py are routed here.
"""
from __future__ import annotations

import enum
import sys
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    pass  # future: typed exc


class SyncDiagnosticCode(str, enum.Enum):
    """Stable, machine-readable codes for final-sync failure classification."""
    LOCK_UNAVAILABLE         = "sync.final_sync_lock_unavailable"
    AUTH_REFRESH_IN_PROGRESS = "sync.auth_refresh_in_progress"
    WEBSOCKET_OFFLINE        = "sync.websocket_offline"
    EVENT_LOOP_UNAVAILABLE   = "sync.event_loop_unavailable"
    SERVER_AUTH_FAILURE      = "sync.server_auth_failure"


# Deduplication: emit each code at most once per process lifetime.
_emitted_codes: set[SyncDiagnosticCode] = set()


def emit_sync_diagnostic(code: SyncDiagnosticCode, message: str) -> None:
    """Write one structured non-fatal diagnostic to stderr.

    At most one line per code is emitted per process invocation.
    Matches the observed format from smoke evidence for backwards compatibility
    with tools that parse this output.
    """
    if code in _emitted_codes:
        return
    _emitted_codes.add(code)
    sys.stderr.write(
        f"sync_diagnostic severity=warning diagnostic_code={code.value} "
        f"fatal=false sync_phase=final_sync message={message}\n"
    )
    sys.stderr.flush()


def classify_sync_error(error_text: str) -> SyncDiagnosticCode:
    """Map a raw error string to a SyncDiagnosticCode.

    Fallback to SERVER_AUTH_FAILURE for unrecognized signals.
    """
    lower = error_text.lower()
    if "lock" in lower and ("unavailable" in lower or "timeout" in lower or "held" in lower):
        return SyncDiagnosticCode.LOCK_UNAVAILABLE
    if "refreshing" in lower or "auth session" in lower or "refresh" in lower and "progress" in lower:
        return SyncDiagnosticCode.AUTH_REFRESH_IN_PROGRESS
    if "websocket" in lower or "ws" in lower and "offline" in lower:
        return SyncDiagnosticCode.WEBSOCKET_OFFLINE
    if "event loop" in lower or "interpreter" in lower or "shutdown" in lower:
        return SyncDiagnosticCode.EVENT_LOOP_UNAVAILABLE
    if "401" in lower or "unauthorized" in lower or "auth" in lower or "token" in lower:
        return SyncDiagnosticCode.SERVER_AUTH_FAILURE
    return SyncDiagnosticCode.SERVER_AUTH_FAILURE


def reset_emitted_codes() -> None:
    """Clear deduplication state. For testing only — do not call in production code."""
    _emitted_codes.clear()
```

**Validation**:
- `SyncDiagnosticCode` has exactly 5 members.
- `emit_sync_diagnostic()` writes to `sys.stderr`, not `sys.stdout`.
- Calling `emit_sync_diagnostic(LOCK_UNAVAILABLE, "...")` twice emits only one line.
- `mypy --strict` passes.

```python
"""Regression tests for sync.diagnostics — FR-001 through FR-006, NFR-003."""
import sys
from unittest.mock import patch, MagicMock
import pytest
from specify_cli.sync.diagnostics import (
    SyncDiagnosticCode,
    emit_sync_diagnostic,
    classify_sync_error,
    reset_emitted_codes,
)


@pytest.fixture(autouse=True)
def clear_dedup():
    reset_emitted_codes()
    yield
    reset_emitted_codes()


def test_lock_unavailable_emitted_to_stderr(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.LOCK_UNAVAILABLE, "lock held")
    captured = capsys.readouterr()
    assert captured.out == ""
    assert "sync.final_sync_lock_unavailable" in captured.err
    assert "fatal=false" in captured.err


def test_auth_refresh_in_progress(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.AUTH_REFRESH_IN_PROGRESS, "refresh in progress")
    captured = capsys.readouterr()
    assert "sync.auth_refresh_in_progress" in captured.err


def test_websocket_offline(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.WEBSOCKET_OFFLINE, "ws offline")
    captured = capsys.readouterr()
    assert "sync.websocket_offline" in captured.err


def test_event_loop_unavailable(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.EVENT_LOOP_UNAVAILABLE, "interpreter shutdown")
    captured = capsys.readouterr()
    assert "sync.event_loop_unavailable" in captured.err


def test_server_auth_failure(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.SERVER_AUTH_FAILURE, "401 from server")
    captured = capsys.readouterr()
    assert "sync.server_auth_failure" in captured.err


def test_deduplication_emits_once(capsys):
    emit_sync_diagnostic(SyncDiagnosticCode.LOCK_UNAVAILABLE, "first call")
    emit_sync_diagnostic(SyncDiagnosticCode.LOCK_UNAVAILABLE, "second call")
    captured = capsys.readouterr()
    assert captured.err.count("sync.final_sync_lock_unavailable") == 1


def test_classify_lock_error():
    assert classify_sync_error("lock unavailable") == SyncDiagnosticCode.LOCK_UNAVAILABLE
    assert classify_sync_error("lock timeout exceeded") == SyncDiagnosticCode.LOCK_UNAVAILABLE


def test_classify_auth_refresh():
    assert classify_sync_error(
        "Another spec-kitty process is refreshing the auth session; retry in a moment"
    ) == SyncDiagnosticCode.AUTH_REFRESH_IN_PROGRESS
```
