## Subtask T012 — Create `support/nested_env.py`

**Purpose**: Reusable helper that creates isolated nested Python environments across uv-managed and standard CPython runners.

**Create** `spec-kitty-end-to-end-testing/support/nested_env.py`:

```python
"""
Reusable helper for creating isolated nested Python environments in E2E scenarios.

Supports uv-managed Python runners (macOS, CI) and standard CPython installs.
"""
from __future__ import annotations

import os
import shutil
import subprocess
import sys
import venv
from dataclasses import dataclass
from pathlib import Path

import pytest


@dataclass
class NestedEnvResult:
    venv_dir: Path
    python: Path    # absolute path to the venv interpreter
    pip: Path       # absolute path to the venv pip
    method: str     # "uv_venv" | "stdlib_venv"


def _is_uv_available() -> bool:
    return shutil.which("uv") is not None


def _try_uv_venv(venv_dir: Path) -> NestedEnvResult | None:
    """Attempt to create a venv using 'uv venv'. Returns None on failure."""
    if not _is_uv_available():
        return None
    result = subprocess.run(
        ["uv", "venv", str(venv_dir), "--python", sys.version[:4]],
        capture_output=True,
        text=True,
    )
    if result.returncode != 0:
        return None
    scripts = "Scripts" if os.name == "nt" else "bin"
    python = venv_dir / scripts / "python"
    pip = venv_dir / scripts / "pip"
    if not python.exists():
        return None
    return NestedEnvResult(venv_dir=venv_dir, python=python, pip=pip, method="uv_venv")


def _try_stdlib_venv(venv_dir: Path) -> NestedEnvResult | None:
    """Attempt to create a venv using stdlib venv. Returns None on failure."""
    try:
        venv.create(venv_dir, with_pip=True, clear=True)
    except Exception:
        return None
    scripts = "Scripts" if os.name == "nt" else "bin"
    python = venv_dir / scripts / "python"
    pip = venv_dir / scripts / "pip"
    if not python.exists():
        return None
    return NestedEnvResult(venv_dir=venv_dir, python=python, pip=pip, method="stdlib_venv")


def create_nested_env(venv_dir: Path) -> NestedEnvResult:
    """
    Create a nested Python environment at venv_dir.

    Prefers uv venv when uv is available (handles uv-managed Python runners).
    Falls back to stdlib venv when safe.
    Raises pytest.skip.Exception with a precise reason if neither method works.
    """
    if venv_dir.exists():
        shutil.rmtree(venv_dir)

    result = _try_uv_venv(venv_dir) or _try_stdlib_venv(venv_dir)

    if result is None:
        uv_available = _is_uv_available()
        reason = (
            "E2E environment limitation: could not create a nested Python environment. "
            f"uv={'available but failed' if uv_available else 'not found'}; "
            "stdlib venv failed (likely uv-managed Python with no libpython). "
            "To run this scenario, ensure a standard CPython installation is on PATH "
            "or that uv is installed and functioning."
        )
        pytest.skip(reason)

    return result
```

```python
"""Unit tests for support.nested_env helper — FR-012, FR-013, FR-014."""
from pathlib import Path
from unittest.mock import MagicMock, patch
import pytest

from support.nested_env import (
    NestedEnvResult,
    _is_uv_available,
    _try_uv_venv,
    _try_stdlib_venv,
    create_nested_env,
)


def test_prefers_uv_when_available(tmp_path):
    venv_dir = tmp_path / "venv"
    mock_result = NestedEnvResult(
        venv_dir=venv_dir,
        python=venv_dir / "bin" / "python",
        pip=venv_dir / "bin" / "pip",
        method="uv_venv",
    )
    with patch("support.nested_env._try_uv_venv", return_value=mock_result) as mock_uv, \
         patch("support.nested_env._try_stdlib_venv") as mock_stdlib:
        result = create_nested_env(venv_dir)
    mock_uv.assert_called_once()
    mock_stdlib.assert_not_called()
    assert result.method == "uv_venv"


def test_stdlib_fallback_when_uv_absent(tmp_path):
    venv_dir = tmp_path / "venv"
    mock_result = NestedEnvResult(
        venv_dir=venv_dir,
        python=venv_dir / "bin" / "python",
        pip=venv_dir / "bin" / "pip",
        method="stdlib_venv",
    )
    with patch("support.nested_env._try_uv_venv", return_value=None), \
         patch("support.nested_env._try_stdlib_venv", return_value=mock_result):
        result = create_nested_env(venv_dir)
    assert result.method == "stdlib_venv"


def test_skip_when_neither_works(tmp_path):
    venv_dir = tmp_path / "venv"
    with patch("support.nested_env._try_uv_venv", return_value=None), \
         patch("support.nested_env._try_stdlib_venv", return_value=None):
        with pytest.raises(pytest.skip.Exception) as exc_info:
            create_nested_env(venv_dir)
    assert "E2E environment limitation" in str(exc_info.value)


def test_result_dataclass_fields(tmp_path):
    venv_dir = tmp_path / "venv"
    mock_result = NestedEnvResult(
        venv_dir=venv_dir,
        python=venv_dir / "bin" / "python",
        pip=venv_dir / "bin" / "pip",
        method="uv_venv",
    )
    with patch("support.nested_env._try_uv_venv", return_value=mock_result):
        result = create_nested_env(venv_dir)
    assert result.venv_dir == venv_dir
    assert result.method in ("uv_venv", "stdlib_venv")
    assert isinstance(result.python, Path)
    assert isinstance(result.pip, Path)
```
