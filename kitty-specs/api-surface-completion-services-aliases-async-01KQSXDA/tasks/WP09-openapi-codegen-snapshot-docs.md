```python
def test_retired_paths_absent_from_openapi_schema(snapshot_spec):
    """Retired alias routes must not appear in the OpenAPI spec paths (FR-014)."""
    paths = snapshot_spec.get("paths", {})
    assert "/api/features" not in paths, (
        "Retired route /api/features must not appear in OpenAPI schema"
    )
    # Check no path matches the kanban pattern
    kanban_paths = [p for p in paths if "/kanban/" in p]
    assert len(kanban_paths) == 0, (
        f"Retired kanban routes must not appear in OpenAPI schema. Found: {kanban_paths}"
    )


def test_no_stale_deprecated_markers_on_retired_routes(snapshot_spec):
    """No path in the spec should have a deprecated:true marker from pre-retirement state."""
    paths = snapshot_spec.get("paths", {})
    deprecated_paths = []
    for path, methods in paths.items():
        for method, operation in methods.items():
            if isinstance(operation, dict) and operation.get("deprecated") is True:
                deprecated_paths.append(f"{method.upper()} {path}")
    # If any routes are intentionally deprecated (not retired), they should be listed here
    # as exceptions. For this mission, all deprecated routes have been retired (410 Gone).
    assert len(deprecated_paths) == 0, (
        f"Found deprecated markers on: {deprecated_paths}. "
        "Retired routes should use include_in_schema=False, not deprecated=True."
    )
```
